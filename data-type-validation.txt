// app/api/decisions/route.ts
import { z } from 'zod';
import { createClient } from '@supabase/supabase-js';
import { NextResponse } from 'next/server';

const ErrorResponse = (code: string, msg: string) => 
  NextResponse.json({ error: { code, message: msg } }, { status: 400 });

const EnforcementSchema = z.object({  // يرفض unknown fields تلقائيًا
  owner_id: z.uuid(),
  requires_approval: z.literal(true),
  approval_deadline: z.iso.datetime(),
  evidence: z.string().max(10000).regex(/^[\s\S]*$/, "must be JSON string").refine(
    (s) => { try { JSON.parse(s); return true; } catch { return false; } },
    "evidence must be valid JSON"
  ).refine(s => !/(?i)(passport|ssn|credit)/.test(s), "PII detected"),
  decision: z.enum(['APPROVED', 'REJECTED', 'ESCALATED']),
  outcome: z.enum(['CLOSED_SUCCESS', 'CLOSED_FAIL', 'PENDING']),
});

export async function POST(req: Request) {
  try {
    const raw = await req.json();
    const body = EnforcementSchema.parse(raw); 

    const supabase = createClient();
    const { error } = await supabase.from('decisions').insert(body);
    if (error) throw error;

    console.info(`[enforce v1.2] write ok, tenant=***, version=2025-03`);

    return NextResponse.json({ success: true });
  } catch (e: any) {
    if (e instanceof z.ZodError) 
      return ErrorResponse('validation_failed', e.errors[0].message);
    return ErrorResponse('enforcement_failed', e.message || 'DB policy violation');
  }
}
